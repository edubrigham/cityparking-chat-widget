Implementatie Code:

<!-- Data Attributes Method (Recommended) -->
<script 
    src="https://your-cdn.com/chat.min.js" 
    data-chat-url="https://your-bot-endpoint.com/bot/your-bot-id" 
    data-btn-position="bottom-left"
    data-widget-btn-color="#ff6b35"
    data-language="nl"
    defer>
</script>

<!-- JavaScript Method (Alternative) -->
<script src="https://your-cdn.com/chat.min.js"></script>
<script>
  window.initCityparkingWidget({
    chatUrl: 'https://your-bot-endpoint.com/bot/your-bot-id',
    language: 'nl',
    theme: {
      primaryColor: '#ff6b35',
      position: 'bottom-left'
    }
  });
</script>
